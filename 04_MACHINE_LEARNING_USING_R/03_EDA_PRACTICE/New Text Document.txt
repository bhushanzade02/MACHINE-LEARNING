my_data <- read.csv("D:/21_ML_STATS/datasaurusdozen-wide.csv")
my_data <- tail(my_data, -1)  
my_data <- data.frame(lapply(my_data, as.numeric))  
par(mar = c(3, 3, 2, 1))
par(mfrow = c(4, 4))
lapply(colnames(my_data)[substr(colnames(my_data), nchar(colnames(my_data)) - 1, nchar(colnames(my_data))) != ".1"], function(col_name) { x <- my_data[[col_name]]; y <- my_data[[paste0(col_name, ".1")]]; model <- lm(y ~ x); plot(x, y, main = paste(col_name), xlab = paste(col_name), ylab = paste(col_name, ".1"), pch = 4, col = "black"); abline(model, col = "red") })
par(mfrow = c(1, 1))
